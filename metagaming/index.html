s = Solver()

flag = []
for i in range(40):
    flag.append(BitVec(f'flag{i}', 32))
    s.add(flag[i] >= 0x20)
    s.add(flag[i] <= 0x7e)

s.add(flag[0] == ord('H'))
s.add(flag[1] == ord('T'))
s.add(flag[2] == ord('B'))
s.add(flag[3] == ord('{'))
s.add(flag[39] == ord('}'))

regs = []
for i in range(15):
    regs.append(BitVecVal(0, 32))
      
def instruction(ins):
    opcode, op1, op2 = ins

    if opcode == 0:
        regs[op1] = flag[op2]
    elif opcode == 1:
        regs[op1] = op2
    elif opcode == 2:
        regs[op1] ^= op2
    elif opcode == 3:
        regs[op1] ^= regs[op2]
    elif opcode == 4:
        regs[op1] |= op2
    elif opcode == 5:
        regs[op1] |= regs[op2]
    elif opcode == 6:
        regs[op1] &= op2
    elif opcode == 7:
        regs[op1] &= regs[op2]
    elif opcode == 8:
        regs[op1] += op2
    elif opcode == 9:
        regs[op1] += regs[op2]
    elif opcode == 10:
        regs[op1] -= op2
    elif opcode == 11:
        regs[op1] -= regs[op2]
    elif opcode == 12:
        regs[op1] *= op2
    elif opcode == 13:
        regs[op1] *= regs[op2]
    elif opcode == 14:
        pass
    elif opcode == 15:
        pass
    elif opcode == 16:
        regs[op1] = RotateRight(regs[op1], z3.BitVecVal(op2,32))
    elif opcode == 17:
        regs[op1] = RotateRight(regs[op1], regs[op2])
    elif opcode == 18:
        regs[op1] = RotateLeft(regs[op1], z3.BitVecVal(op2,32))
    elif opcode == 19:
        regs[op1] = RotateLeft(regs[op1], regs[op2])
    elif opcode == 20:
        regs[op1] = regs[op2]
    elif opcode == 21:
        regs[op1] = 0
    elif opcode == 22:
        regs[op1] >>= op2
    elif opcode == 23:
        regs[op1] >>= regs[op2]
    elif opcode == 24:
        regs[op1] <<= op2
    elif opcode == 25:
        regs[op1] <<= regs[op2]
    else:
        assert False

for ins in payload:
    instruction(ins)
